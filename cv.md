### Code example: :point_down:

** Find the odd int ** : Given an array of integers, find the one that appears an odd number of times. There will always be only one integer that appears an odd number of times.

```
function findOdd(A) {
  let obj = {}, i = 0, ans;

  A.forEach((num) => {
      if (!obj[num]) {
        obj[num] = ++i;
      } else {
        obj[num] += 1;
      }
      i = 0;
  });

  for (let i in obj) {
      if (obj[i] % 2 == 1) {
          ans = i;
          return Number(i);
      }
  }
}
```
